import { Meta, ColorPalette, ColorItem, Heading } from "@storybook/addon-docs/blocks";
import colors from "@reservando/design-system/styles/_colors.scss";
const sections = {};
Object.keys(colors).forEach((color) => {
  const section = color.split("-")[0];
  if (!sections[section]) {
    sections[section] = {};
  }
  sections[section][color] = colors[color];
});

<Meta title="Introduction/Colors" />

# Colors

<Fragment>
  {Object.keys(sections).map((section) => {
    return (
      <Fragment>
        <Heading>{section.charAt(0).toUpperCase() + section.slice(1)}</Heading>
        <br />
        {Object.keys(sections[section]).map((color) => (
          <ColorItem subtitle={`$${color}`} colors={[sections[section][color]]} />
        ))}
      </Fragment>
    );
  })}
</Fragment>
